00374 bool IAITableTopControllerModule::updateModule() {
00375     using std::cout;
00376     using yarp::os::Network;
00377 
00378     // Check for running threads
00379     // Posreader thread
00380     if (!thPosRead->isRunning()) {
00381         cerr << dbgTag << "PositionRead thread is not running. Stopping the module. \n";
00382         return false;
00383     }
00384     // FTSensorReader thread
00385     if (!thFTRead->isRunning()) {
00386         cerr << dbgTag << "FTSensorRead thread is not running. Stopping the module. \n";
00387         return false;
00388     }
00389 
00390 
00391     // Run experiment
00392     if (stepCounter < 0) {
00393         // Wait for 5 seconds before starting the experiment
00394         Time::delay(5.0);
00395 
00396         // Connect datadumper
00397         Network::connect("/IAITableTop/posreader/position:o", "/dump_iaittpos");
00398         Network::connect(portIAITTControllerOutExperimentStatus.getName(), "/dump_iaittexp");
00399         Network::connect("/NIDAQmxReader/data/real:o", "/dump_nano17");
00400         Network::connect("/SkinTableTop/skin/fingertip", "/dump_fingertip");
00401 
00402         // Increment step counter: experiment is ready to be run
00403         stepCounter++;
00404     } else if ((stepCounter >= 0) && (stepCounter < nSteps)) {
00405         cout << "\n" << dbgTag << "Performing experiment step n. " << stepCounter << "\n";
00406         
00407         // Get step position
00408         Bottle cmdX, cmdY, cmdZ;
00409         buildMoveAbsX(stepCounter, cmdX);
00410         buildMoveAbsY(stepCounter, cmdY);
00411         buildMoveAbsZ(stepCounter, cmdZ);

00413         // Move to step position
00414         cout << dbgTag << "Moving to X position. \n";
00415         if(!sendCommandToSerial(cmdX)) {
00416             cerr << dbgTag << "Could not send movement command to robot. \n";
00417             return false;
00418         }
00419             
00420         cout << dbgTag << "Moving to Y position. \n";
00421         if(!sendCommandToSerial(cmdY)) {
00422             cerr << dbgTag << "Could not send movement command to robot. \n";
00423             return false;
00424         }
00425             
00426         cout << dbgTag << "Moving to Z position. \n";    
00427         if(!sendCommandToSerial(cmdZ)) {
00428             cerr << dbgTag << "Could not send movement command to robot. \n";
00429             return false;
00430         }
00431 
00432         // Wait before reading robot status. This ensures the retrieved status is up to date.
00433         Time::delay(statusReadDelay);
00434 
00435         // Wait for movement to be over
00436         if(!waitMoveDone(0.05, 10)) {
00437             cerr << dbgTag << "Could not complete the movement. \n";
00438             return false;
00439         }
00440 
00441         // Stay in reached position for the selected time
00442         double timeStart = Time::now();
00443         while(Time::now() - timeStart <= timeIntervals[stepCounter]) {
00444             Time::delay(0.01);
00445         }
00446         double timeEnd = Time::now();
00447 
00448         // Print out step time
00449         double deltaT = timeEnd - timeStart;
00450         cout  << dbgTag << "Elapsed time: " << deltaT << "s. \n";
00451 
00452         // Write out experiment status
00453         writeExperimentStatus(timeStart, timeEnd);
00454 
00455         // Increment step counter
00456         stepCounter++;
00457     } else {
00458         cout << dbgTag << "Experiment completed. \n";
00459 
00460         // Disconnect datadumper
00461         Network::disconnect("/IAITableTop/posreader/position:o", "/dump_iaittpos");
00462         Network::disconnect(portIAITTControllerOutExperimentStatus.getName(), "/dump_iaittexp");

00551 /* ******* Waits for the movement to be complete.                           ********************************************** */
00552 bool IAITableTopControllerModule::waitMoveDone(const double &i_period, const double &i_timeout) {
00553     using std::string;
00554     using yarp::sig::Vector;
00555 
00556     bool done = false;
00557 
00558     double tStart = Time::now();
00559     while (!done) {
00560         // Check for timeout
00561         if (Time::now() - tStart > i_timeout) {
00562             cerr << dbgTag << "Movement completion check timeout expired. \n";
00563             return false;
00564         }
00565 
00566         // Read axis status
00567         Vector *reply = portIAITTPositionReadInStatus.read(true);
00568         
00569         if ((reply) && (reply->size() > 0)) {
00570             // Check axis status - 0x1C == free
00571             done = (((*reply)[0] == 0x1C)
00572                 && ((*reply)[1] == 0x1C)
00573                 && ((*reply)[2] == 0x1C));
00574 
00575 #if IAITABLETOP_CONTROLLER_DEBUG
00576             cout << dbgTag << "DEBUG: Axis status is: \t";
00577             for (size_t i = 0; i < reply->size(); ++i) {
00578                 cout << std::hex << std::showbase << std::uppercase << (int) (*reply)[i] << " ";
00579             }
00580             cout << std::dec << std::noshowbase << std::nouppercase << "\n";
00581 #endif
00582 
00583         } else {
00584 //            cerr << dbgTag << "There was an error reading from the axis status port. \n";
00585             return false;
00586         }
00587 
00588         // Wait before reading checking again
00589         Time::delay(i_period);
00590     }
00591 
00592     std::cout << dbgTag << "TableTop: Movement complete." << "\n";
00593 
00594     return done;
00595 }
