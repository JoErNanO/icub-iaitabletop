00076 /* ******* Run thread                                                       ********************************************** */
00077 void IAITableTopFTSensorReadThread::run(void) {
00078     using std::deque;
00079     using yarp::sig::Vector;
00080     using yarp::os::Bottle;
00081 
00082     // Read sensor data from port
00083     Vector *data = portFTSensorReaderInData.read(false);
00084     if ((data) && (data->size() > 0)) {
00085 #if IAITABLETOP_TH_FTREAD_DEBUG
00086         // Print out FT sensor data
00087         cout << dbgTag << "FT sensor data: ";
00088         for (size_t i = 0; i < data->size(); ++i) {
00089             cout << (*data)[i] << " ";
00090         }
00091         cout << "\n";
00092 #endif
00093 
00094         // Check sensor data size against safety threshold size
00095         if (data->size() != safetyThresholds.size()) {
00096             cerr << dbgTag << "The FT sensor data and the safety threshold vector do not have the same size. The FT sensor data size is " << data->size()
00097                 << " , while the safety threshold vector size is " << safetyThresholds.size() << ". Check the module configuration file. \n";
00098             cerr << dbgTag << "An error occurred so the thread will now stop. \n";
00099             askToStop();
00100         } else {
00101             // Check if values are higher than their respective thresholds
00102             deque<bool> flags;
00103             if (checkFTValues(data, flags)) {
00104                 cout << dbgTag << "FT sensor safety threshold reached. \n";
00105                 serialMutex->wait();
00106                 
00107                 // Send stop command to all axes
00108                 Bottle cmd, reply;
00109                 cmd.addString("!992380701@@\n");
00110                 iSerialPort->send(cmd);
00111 
00112 #ifdef _WIN32
00113                 // Wait for a small delay to avoid reading incomplete replies
00114                 Time::delay(0.05);
00115 #endif
00116                 iSerialPort->receive(reply);
00117 
00118                 serialMutex->post();
00119 
00120                 cerr << dbgTag << "The FT sensor recorded a value which is higher than its safety threshold. The robot will now be stopped. \n";
00121                 askToStop();
00122             }
00123         }
00124     }
00125 }

00147 /* ******* Check if FT sensor values are higher than the threshold.         ********************************************** */
00148 bool IAITableTopFTSensorReadThread::checkFTValues(yarp::sig::Vector * const i_vals, std::deque<bool> &o_reachedLimit) {
00149     bool thresholdReached = false;
00150     
00151     o_reachedLimit.resize(i_vals->size());
00152     // Check all FT values
00153     for (size_t i = 0; i < i_vals->size(); ++i) {
00154         o_reachedLimit[i] = (std::abs((*i_vals)[i]) >= safetyThresholds[i]);
00155 
00156         if (!thresholdReached) {
00157             thresholdReached = o_reachedLimit[i];
00158         }
00159     }
00160 
00161     return thresholdReached;
00162 }
